1. Run the rbp_search_for_PHAROKKA_phage_annotations.py (instructions within the python script)
2. Find and install the BLAST virus datasets at the location of your choosing through:
wget https://ftp.ncbi.nlm.nih.gov/refseq/release/viral/viral.1.protein.faa.gz
gunzip viral.1.protein.faa.gz 
3. Run the following to make the database:
makeblastdb     -in viral.1.protein.faa     -dbtype prot     -out viral_db     -parse_seqids
4. Run the following to BLAST the results obtained from step 1:
for faa in *_rbp.faa; do

    out="${faa%.faa}_blast.tsv"

    if [ -s "$out" ]; then
        echo "Skipping $faa (already done)"
        continue
    fi

    echo "Running BLAST for $faa"

    blastp \
        -query "$faa" \
        -db /home/deces123/Project/Vin_Phage/blast_db_phage/viral_db \
        -max_target_seqs 2 \
        -evalue 1e-5 \
        -outfmt "6 qseqid sseqid pident length evalue bitscore stitle" \
        -out "$out"

    sleep 5

done
